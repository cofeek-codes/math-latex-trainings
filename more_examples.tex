\begin{center}
	Example 1:
\end{center}

\begin{equation}
	\int\limits_{0}^{\frac{\pi}{2}} \cos{x} + 5^{x} = \sin{x} + \frac{5x^{2}}{2}
\end{equation}

\begin{center}
	Example 2:
\end{center}

\begin{equation}
	\int\limits_{0}^{\frac{\pi}{4}} \frac{4}{\cos^{2}{x}} + \frac{1}{\pi} = 4 \tan{x} + \frac{1}{\pi}x
\end{equation}

\begin{center}

\begin{equation}
	\int\limits_{0}^{\frac{\pi}{4}} = (4\tan{\frac{\pi}{4}}+\frac{1}{\pi}) - (4 \tan{0} + \frac{1}{\pi}0) = \frac{17}{4}
\end{equation}
